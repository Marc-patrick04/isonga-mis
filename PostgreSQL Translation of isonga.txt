-- Function to generate case number
CREATE OR REPLACE FUNCTION generate_case_number()
RETURNS TRIGGER AS $$
DECLARE
    next_num INTEGER;
    current_year INTEGER;
BEGIN
    IF NEW.case_number IS NULL THEN
        current_year := EXTRACT(YEAR FROM CURRENT_DATE);
        SELECT COALESCE(MAX(CAST(SUBSTRING(case_number FROM 10) AS INTEGER)), 0) + 1 INTO next_num
        FROM arbitration_cases
        WHERE EXTRACT(YEAR FROM created_at) = current_year;
        NEW.case_number := CONCAT('ARB-', current_year, '-', LPAD(next_num::TEXT, 4, '0'));
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER before_case_insert
    BEFORE INSERT ON arbitration_cases
    FOR EACH ROW
    EXECUTE FUNCTION generate_case_number();

-- Function for committee_audit trigger
CREATE OR REPLACE FUNCTION after_committee_audit_insert()
RETURNS TRIGGER AS $$
DECLARE
    target_user_id INTEGER;
BEGIN
    SELECT user_id INTO target_user_id 
    FROM committee_members 
    WHERE id = NEW.target_member_id;
    
    IF target_user_id IS NOT NULL THEN
        INSERT INTO system_notifications (
            user_id,
            notification_type,
            title,
            message,
            related_id,
            related_table,
            created_by,
            expires_at
        ) VALUES (
            target_user_id,
            'audit_request',
            'Audit Request from President of Representative Board',
            CONCAT('You have been requested to undergo a ', NEW.audit_type, ' audit for the period ', 
                   TO_CHAR(NEW.audit_period, 'Month YYYY'), '. Please check the Committee Audit section for details.'),
            NEW.id,
            'committee_audit',
            NEW.audited_by,
            CURRENT_TIMESTAMP + INTERVAL '7 days'
        );
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER after_committee_audit_insert_trigger
    AFTER INSERT ON committee_audit
    FOR EACH ROW
    EXECUTE FUNCTION after_committee_audit_insert();

-- Function for after_vote_insert trigger
CREATE OR REPLACE FUNCTION after_vote_insert()
RETURNS TRIGGER AS $$
BEGIN
    UPDATE election_voters 
    SET has_voted = TRUE, voted_at = NOW() 
    WHERE id = NEW.voter_id;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER after_vote_insert_trigger
    AFTER INSERT ON election_votes
    FOR EACH ROW
    EXECUTE FUNCTION after_vote_insert();

-- View: active_arbitration_cases
CREATE VIEW active_arbitration_cases AS
SELECT 
    ac.id,
    ac.case_number,
    ac.title,
    ac.description,
    ac.case_type,
    ac.complainant_id,
    ac.respondent_id,
    ac.complainant_name,
    ac.respondent_name,
    ac.complainant_contact,
    ac.respondent_contact,
    ac.priority,
    ac.status,
    ac.assigned_to,
    ac.filing_date,
    ac.hearing_date,
    ac.resolution_date,
    ac.resolution_details,
    ac.created_by,
    ac.created_at,
    ac.updated_at,
    u.full_name AS assigned_to_name,
    (CURRENT_DATE - ac.filing_date) AS days_open
FROM arbitration_cases ac
LEFT JOIN users u ON ac.assigned_to = u.id
WHERE ac.status IN ('filed', 'under_review', 'hearing_scheduled', 'mediation')
ORDER BY 
    CASE ac.priority
        WHEN 'urgent' THEN 1
        WHEN 'high' THEN 2
        WHEN 'medium' THEN 3
        WHEN 'low' THEN 4
    END ASC,
    ac.filing_date DESC;

-- View: election_results
CREATE VIEW election_results AS
SELECT 
    ec.election_id,
    ec.position,
    ec.student_name,
    ec.reg_number,
    COUNT(ev.id) AS vote_count,
    ROUND((COUNT(ev.id) * 100.0) / NULLIF(
        (SELECT COUNT(*) 
         FROM election_votes ev2 
         JOIN election_candidates ec2 ON ev2.candidate_id = ec2.id 
         WHERE ec2.position = ec.position AND ec2.election_id = ec.election_id), 0), 2) AS percentage
FROM election_candidates ec
LEFT JOIN election_votes ev ON ec.id = ev.candidate_id
GROUP BY ec.id, ec.election_id, ec.position, ec.student_name, ec.reg_number
ORDER BY ec.election_id, ec.position, vote_count DESC;

-- View: election_statistics
CREATE VIEW election_statistics AS
SELECT 
    e.id,
    e.title,
    e.academic_year,
    e.status,
    COUNT(DISTINCT ec.id) AS candidate_count,
    COUNT(DISTINCT ev.id) AS voter_count,
    COUNT(DISTINCT ev2.id) AS votes_cast,
    ROUND((COUNT(DISTINCT ev2.id) * 100.0) / NULLIF(COUNT(DISTINCT ev.id), 0), 2) AS turnout_percentage
FROM elections e
LEFT JOIN election_candidates ec ON e.id = ec.election_id
LEFT JOIN election_voters ev ON e.id = ev.election_id
LEFT JOIN election_voters ev2 ON e.id = ev2.election_id AND ev2.has_voted = TRUE
GROUP BY e.id, e.title, e.academic_year, e.status;

-- View: news_with_categories

-- Procedure: CalculateElectionResults
CREATE OR REPLACE PROCEDURE CalculateElectionResults(election_id_param INTEGER)
LANGUAGE plpgsql
AS $$
BEGIN
    SELECT 
        position,
        student_name,
        reg_number,
        vote_count,
        percentage,
        CASE 
            WHEN vote_count = MAX(vote_count) OVER (PARTITION BY position) THEN 'WINNER'
            ELSE ''
        END as result
    FROM election_results 
    WHERE election_id = election_id_param
    ORDER BY position, vote_count DESC;
END;
$$;

-- Procedure: GetArbitrationStatistics
CREATE OR REPLACE PROCEDURE GetArbitrationStatistics(start_date DATE, end_date DATE)
LANGUAGE plpgsql
AS $$
BEGIN
    SELECT 
        status,
        COUNT(*) as case_count,
        AVG(EXTRACT(DAY FROM (COALESCE(resolution_date, CURRENT_DATE) - filing_date))) as avg_resolution_days
    FROM arbitration_cases 
    WHERE filing_date BETWEEN start_date AND end_date
    GROUP BY status;
END;
$$;

-- Procedure: IncrementNewsViews
CREATE OR REPLACE PROCEDURE IncrementNewsViews(news_id INTEGER)
LANGUAGE plpgsql
AS $$
BEGIN
    UPDATE news SET views_count = views_count + 1 WHERE id = news_id;
END;
$$;
